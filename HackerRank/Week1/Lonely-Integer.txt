 public static int lonelyinteger(List<Integer> a) {
    // Write your code here
    Collections.sort(a);
    for(int i=0;i<a.size()-1;i+=2){
        if(a.get(i)==a.get(i+1)){
            continue;
        }
        else{
            return a.get(i);
        }
        
    }
    return a.get(a.size()-1);

    }
